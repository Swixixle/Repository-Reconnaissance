L11: /**
L12:  * Reorders audio chunks that may arrive out of sequence.
L13:  * Buffers chunks until they can be played in correct order.
L14:  *
L15:  * Example: If chunks arrive as seq 2, seq 0, seq 1:
L16:  * - seq 2 arrives → buffered (waiting for seq 0)
L17:  * - seq 0 arrives → played immediately, then check buffer
L18:  * - seq 1 arrives → played immediately (seq 0 done), seq 2 now plays
L19:  */

L21:   private pending = new Map<number, string[]>();
L22:   private nextSeq = 0;
L23: 
L24:   /** Add chunk with sequence number, returns chunks ready to play in order */
L25:   push(seq: number, data: string): string[] {
L26:     // Store the chunk under its sequence number
L27:     if (!this.pending.has(seq)) {
L28:       this.pending.set(seq, []);
L29:     }
L30:     this.pending.get(seq)!.push(data);
L31: 
L32:     // Drain consecutive ready sequences
L33:     const ready: string[] = [];
L34:     while (this.pending.has(this.nextSeq)) {
L35:       ready.push(...this.pending.get(this.nextSeq)!);
L36:       this.pending.delete(this.nextSeq);
L37:       this.nextSeq++;
L38:     }
L39:     return ready;
L40:   }
L41: 
L42:   reset() {
L43:     this.pending.clear();
L44:     this.nextSeq = 0;
L45:   }
L46: }
